1     ! INFNAN is a module defining some procedures concerning INF or NAN.
2     !
3     ! Here we implement the procedures for single, double, and quadruple
4     ! precisions, because when we interface the Fortran code with other
5     ! languages (e.g., MATLAB), the procedures may be invoked in both the 
6     ! Fortran code and the gateway (e.g., MEX gateway), which may use
7     ! different real precisions (e.g., the Fortran code may use single 
8     ! precision, but MEX gateway uses double precision by default).
9     !
10     ! Coded by Zaikun Zhang in July 2020.
11

19     #endif

70     ! The following functions check whether a real number x is infinite, 
71     ! nan, or finite. Starting from Fortran 2003, the intrincic 
72     ! ieee_arithmetic module provides ieee_is_nan() and ieee_is_finite().
73     
74     
75     #if __USE_IEEE_ARITHMETIC__ == 0 
76     pure elemental function is_nan_sp(x) result(y)
77     use consts_mod, only : SP 
78     implicit none
79     real(SP), intent(in) :: x
80     logical :: y
81     y = (.not. (x >= x))
82     ! This implementation may be more costly than y = (x /= x).
83     ! However, if we defined is_nan = (x /= x), gfortran will complain
84     ! about inequality comparison between floating-point numbers.
85     end function is_nan_sp
86     
87     pure elemental function is_nan_dp(x) result(y)
88     use consts_mod, only : DP 
89     implicit none
90     real(DP), intent(in) :: x
91     logical :: y
92     y = (.not. (x >= x))
93     end function is_nan_dp
94     
95     pure elemental function is_finite_sp(x) result(y)
96     use consts_mod, only : SP
97     implicit none
98     real(SP), intent(in) :: x
99     logical :: y
100     y = (abs(x) <= huge(x))
101     end function is_finite_sp
102     
103     pure elemental function is_finite_dp(x) result(y)
104     use consts_mod, only : DP
105     implicit none
106     real(DP), intent(in) :: x
107     logical :: y
108     y = (abs(x) <= huge(x))
109     end function is_finite_dp
110     
111     #endif
