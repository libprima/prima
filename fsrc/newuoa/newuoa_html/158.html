18     subroutine updateh(knew, beta, vlag_in, idz, bmat, zmat)
19     ! UPDATE updates arrays BMAT and ZMAT together with IDZ, in order
20     ! to replace the interpolation point XPT(:, KNEW) by XNEW. On entry,
21     ! VLAG_IN contains the components of the vector THETA*WCHECK + e_b
22     ! of the updating formula (6.11) in the NEWUOA paper, and BETA
23     ! holds the value of the parameter that has this name. VLAG_IN and BETA
24     ! contains information about XNEW, because they are calculated according
25     ! to D = XNEW - XOPT.
26     !
27     ! See Section 4 of the NEWUOA paper.
28     
29     ! Generic modules
30     use consts_mod, only : RP, IK, ONE, ZERO, DEBUGGING, SRNLEN
31     use debug_mod, only : errstop, verisize
32     use lina_mod, only : grota, r2update, symmetrize
33     
34     implicit none
35     
36     ! Inputs
37     integer(IK), intent(in) :: knew
38     real(RP), intent(in) :: beta
39     real(RP), intent(in) :: vlag_in(:)     ! VLAG_IN(NPT + N)
40     
41     ! In-outputs
42     integer(IK), intent(inout) :: idz
43     real(RP), intent(inout) :: bmat(:, :)  ! BMAT(N, NPT + N)
44     real(RP), intent(inout) :: zmat(:, :)  ! ZMAT(NPT, NPT - N - 1)
45     
46     ! Intermediate variables
47     integer(IK) :: j
48     integer(IK) :: ja
49     integer(IK) :: jb
50     integer(IK) :: jl
51     integer(IK) :: n
52     integer(IK) :: npt
53     real(RP) :: alpha
54     real(RP) :: denom
55     real(RP) :: scala
56     real(RP) :: scalb
57     real(RP) :: sqrtdn
58     real(RP) :: tau
59     real(RP) :: tausq
60     real(RP) :: temp
61     real(RP) :: tempa
62     real(RP) :: tempb
63     real(RP) :: v1(size(bmat, 1))
64     real(RP) :: v2(size(bmat, 1))
65     real(RP) :: vlag(size(vlag_in))  ! Copy of VLAG_IN
66     real(RP) :: w(size(vlag))
67     real(RP) :: ztemp(size(zmat, 1))
68     logical :: reduce_idz
69     character(len = SRNLEN), parameter :: srname = 'UPDATEH'
70     
71     
72     ! Get and verify the sizes.
73     n = int(size(bmat, 1), kind(n))
74     npt = int(size(bmat, 2), kind(npt)) - n
75     
76     if (DEBUGGING) then
77         if (n == 0 .or. npt < n + 2) then
78             call errstop(srname, 'SIZE(BMAT) is invalid')
79         end if
80         call verisize(zmat, npt, int(npt - n - 1, kind(n)))
81         call verisize(vlag, npt + n)
82     end if
83     
84     vlag = vlag_in  ! VLAG_IN is INTENT(IN) and cannot be revised.
85     
86     ! Apply the rotations that put zeros in the KNEW-th row of ZMAT.
87     ! A Givens rotation will be multiplied to ZMAT from the left so
88     ! ZMAT(KNEW, JL) becomes SQRT(ZMAT(KNEW, JL)^2 + ZMAT(KNEW, J)) and
89     ! ZMAT(KNEW, J) becomes 0.
90     jl = 1  ! For J = 2, ..., IDZ - 1, set JL = 1.
91     do j = 2, int(idz - 1, kind(j))
92         call grota(zmat, jl, j, knew)
93     end do
94     if (idz <= npt - n - 1) then
95         jl = idz  ! For J = IDZ + 1, ..., NPT - N - 1, set JL = IDZ.
96     end if
97     do j = int(idz + 1, kind(j)), int(npt - n - 1, kind(j))
98         call grota(zmat, jl, j, knew)
99     end do
100     
101     ! JL plays an important role below. Its value is determined by the
102     ! current (i.e., unupdated) value of IDZ. IDZ is an integer in
103     ! {1, ..., NPT - N} such that s_j = -1 for j < IDZ while s_j = 1
104     ! for j >= IDZ in the factorization of Omega. See (3.17), (4.16)
105     ! of the NEWUOA paper.
106     !
107     ! For the value of JL, there are two possibilities:
108     ! 1. JL = 1 iff IDZ = 1 or IDZ = NPT - N.
109     ! 1.1. IDZ = 1 means that
110     ! Omega = sum_{J=1}^{NPT-N-1} ZMAT(:, J)*ZMAT(:, J)' ;
111     ! 1.2. IDZ = NPT - N means that
112     ! Omega = - sum_{J=1}^{NPT-N-1} ZMAT(:, J)*ZMAT(:, J)' ;
113     ! 2. JL = IDZ > 1 iff 2 <= IDZ <= NPT - N - 1.
114     
115     ! Put the first NPT components of the KNEW-th column of HLAG into
116     ! W, and calculate the parameters of the updating formula.
117     tempa = zmat(knew, 1)
118     if (idz >=  2) then
119         tempa = -tempa
120     end if
121     
122     w(1 : npt) = tempa*zmat(:, 1)
123     if (jl > 1) then
124         tempb = zmat(knew, jl)
125         w(1 : npt) = w(1 : npt) + tempb*zmat(:, jl)
126     end if
127     
128     alpha = w(knew)
129     tau = vlag(knew)
130     tausq = tau*tau
131     denom = alpha*beta + tausq
132     ! After the following line, VLAG = Hw - e_t in the NEWUOA paper.
133     vlag(knew) = vlag(knew) - ONE
134     sqrtdn = sqrt(abs(denom))
135     
136     ! Complete the updating of ZMAT when there is only one nonzero
137     ! element in the KNEW-th row of the new matrix ZMAT, but, if
138     ! IFLAG is set to one, then the first column of ZMAT will be
139     ! exchanged with another one later.
140     reduce_idz = .false.
141     if (jl == 1) then
142         ! There is only one nonzero in ZMAT(KNEW, :) after the
143         ! rotation. This is the normal case, because IDZ is 1 in
144         ! precise arithmetic.
145         !------------------------------------------------------------!
146         ! Up to now, TEMPA = ZMAT(KNEW, 1) if IDZ = 1 and
147         ! TEMPA = -ZMAT(KNEW, 1) if IDZ >= 2. However, according to
148         ! (4.18) of the NEWUOA paper, TEMPB should always be
149         ! ZMAT(KNEW, 1)/sqrtdn regardless of IDZ. Therefore, the
150         ! following definition of TEMPB is inconsist with (4.18). This
151         ! is probably a BUG. See also Lemma 4 and (5.13) of Powell's
152         ! paper "On updating the inverse of a KKT matrix". However,
153         ! the inconsistency is hardly observable in practice, because
154         ! JL = 1 implies IDZ = 1 in precise arithmetic.
155         !++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++!
156         !tempb = tempa/sqrtdn
157         !tempa = tau/sqrtdn
158         !++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++!
159         ! Here is the corrected version (only TEMPB is changed).
160         tempa = tau/sqrtdn
161         tempb = zmat(knew, 1)/sqrtdn
162         !------------------------------------------------------------!
163     
164         zmat(:, 1) = tempa*zmat(:, 1) - tempb*vlag(1 : npt)
165     
166         !------------------------------------------------------------!
167         ! The following six lines by Powell are obviously problematic
168         ! --- TEMP is always nonnegative.  According to (4.18) of the
169         ! NEWUOA paper, the "TEMP < ZERO" and "TEMP >= ZERO" below
170         ! should be both revised to "DENOM < ZERO". See also the
171         ! corresponding part of the LINCOA code. Note that the NEAUOA
172         ! paper uses SIGMA to denote DENOM. Check also Lemma 4 and
173         ! (5.13) of Powell's paper "On updating the inverse of a KKT
174         ! matrix". It seems that the BOBYQA code does not have this
175         ! part --- it does not have IDZ at all (why?). Anyway, these
176         ! lines are not invoked very often in practice, because IDZ
177         ! should always be 1 in precise arithmetic.
178         !++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++!
179         !if (idz == 1 .and. sqrtdn < ZERO) then
180         !    idz = 2
181         !end if
182         !if (idz >= 2 .and. sqrtdn >= ZERO) then
183         !    reduce_idz = .true.
184         !end if
185         !++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++!
186         ! This is the corrected version. It copies precisely the
187         ! corresponding part of the LINCOA code.
188         if (denom < ZERO) then
189             if (idz == 1) then
190             ! This is the first place (out of two) where IDZ is
191             ! increased. Note that IDZ = 2 <= NPT-N after the update.
192                 idz = 2
193             else
194             ! This is the first place (out of two) where IDZ is
195             ! decreased (by 1). Since IDZ >= 2 in this case, we have
196             ! IDZ >= 1 after the update.
197                 reduce_idz = .true.
198             end if
199         end if
200         !------------------------------------------------------------!
201     else
202         ! Complete the updating of ZMAT in the alternative case.
203         ! There are two nonzeros in ZMAT(KNEW, :) after the rotation.
204         ja = 1
205         if (beta >=  ZERO) then
206             ja = jl
207         end if
208         jb = int(jl + 1 - ja, kind(jb))
209         temp = zmat(knew, jb)/denom
210         tempa = temp*beta
211         tempb = temp*tau
212         temp = zmat(knew, ja)
213         scala = ONE/sqrt(abs(beta)*temp*temp + tausq)
214         scalb = scala*sqrtdn
215         zmat(:, ja) = scala*(tau*zmat(:, ja) - temp*vlag(1 : npt))
216         zmat(:, jb) = scalb*(zmat(:, jb) - tempa*w(1 : npt) - tempb*vlag(1 : npt))
217         ! If and only if DENOM <= 0, IDZ will be revised according
218         ! to the sign of BETA. See (4.19)--(4.20) of the NEWUOA paper.
219         if (denom <=  ZERO) then
220             if (beta < ZERO) then
221             ! This is the second place (out of two) where IDZ is
222             ! increased. Since JL = IDZ <= NPT-N-1 in this case,
223             ! we have IDZ <= NPT-N after the update.
224                 idz = int(idz + 1, kind(idz))
225             end if
226             if (beta >=  ZERO) then
227             ! This is the second place (out of two) where IDZ is
228             ! decreased (by 1). Since IDZ >= 2 in this case, we have
229             ! IDZ >= 1 after the update.
230                 reduce_idz = .true.
231             end if
232         end if
233     end if
234     
235     ! IDZ is reduced in the following case, and usually the first
236     ! column of ZMAT is exchanged with a later one.
237     if (reduce_idz) then
238         idz = int(idz - 1, kind(idz))
239         if (idz > 1) then
240             ztemp = zmat(:, 1)
241             zmat(:, 1) = zmat(:, idz)
242             zmat(:, idz) = ztemp
243         end if
244     end if
245     
246     ! Finally, update the matrix BMAT.
247     w(npt + 1 : npt + n) = bmat(:, knew)
248     v1 = (alpha*vlag(npt+1 : npt+n) - tau*w(npt+1 : npt+n))/denom
249     v2 = (-beta*w(npt+1 : npt+n) - tau*vlag(npt+1 : npt+n))/denom
250     
251     call r2update(bmat, ONE, v1, vlag, ONE, v2, w)
252     ! In floating-point arithmetic, the update above does not guarante
253     ! BMAT(:, NPT+1 : NPT+N) to be symmetric. Symmetrization needed.
254     call symmetrize(bmat(:, npt + 1 : npt + n))
255     
256     end subroutine updateh

259     subroutine updateq(idz, knew, bmatknew, fqdiff, zmat, xptknew, gq, hq, pq)
260     ! UPDATEQ updates GQ, HQ, and PQ when XPT(:, KNEW) is replaced by XNEW.
261     ! See Section 4 of the NEWUOA paper.
262     
263     ! Generic modules
264     use consts_mod, only : RP, IK, ZERO, DEBUGGING, SRNLEN
265     use debug_mod, only : errstop, verisize
266     use lina_mod, only : r1update, Ax_plus_y
267     
268     implicit none
269     
270     ! Inputs
271     integer(IK), intent(in) :: idz
272     integer(IK), intent(in) :: knew
273     real(RP), intent(in) :: bmatknew(:) ! BMATKNEW(N)
274     ! fqdiff = [f(xnew) - f(xopt)] - [q(xnew) - q(xopt)] = moderr
275     real(RP), intent(in) :: fqdiff 
276     real(RP), intent(in) :: zmat(:, :)  ! ZMAT(NPT, NPT - N - 1)
277     real(RP), intent(in) :: xptknew(:)  ! XPTKNEW(N)
278     
279     ! In-outputs
280     real(RP), intent(inout) :: gq(:)   ! GQ(N)
281     real(RP), intent(inout) :: hq(:, :)! HQ(N, N)
282     real(RP), intent(inout) :: pq(:)   ! PQ(NPT)
283     
284     ! Intermediate variables
285     integer(IK) :: n
286     integer(IK) :: npt
287     real(RP) :: fqdz(size(zmat, 2))
288     character(len = SRNLEN), parameter :: srname = 'UPDATEQ'
289     
290     
291     ! Get and verify the sizes.
292     n = int(size(gq), kind(n))
293     npt = int(size(pq), kind(npt))
294     
295     if (DEBUGGING) then
296         if (n == 0 .or. npt < n + 2) then
297             call errstop(srname, 'SIZE(GQ) or SIZE(PQ) is invalid')
298         end if
299         call verisize(zmat, npt, int(npt - n - 1, kind(n)))
300         call verisize(xptknew, n)
301         call verisize(bmatknew, n)
302         call verisize(hq, n, n)
303     end if
304     
305     !----------------------------------------------------------------!
306     ! Implement R1UPDATE properly so that it ensures HQ is symmetric.
307     call r1update(hq, pq(knew), xptknew)
308     !----------------------------------------------------------------!
309     
310     ! Update the implicit part of second derivatives.
311     fqdz = fqdiff*zmat(knew, :)
312     fqdz(1 : idz - 1) = -fqdz(1 : idz - 1)
313     pq(knew) = ZERO
314     !----------------------------------------------------------------!
315     !pq = pq + matprod(zmat, fqdz) !---------------------------------!
316     pq = Ax_plus_y(zmat, fqdz, pq)
317     !----------------------------------------------------------------!
318     
319     ! Update the gradient.
320     gq = gq + fqdiff*bmatknew
321     
322     end subroutine updateq

325     subroutine tryqalt(idz, fval, ratio, smat, zmat, itest, gq, hq, pq)
326     ! TRYQALT tests whether to replace Q by the alternative model,
327     ! namely the model that minimizes the F-norm of the Hessian
328     ! subject to the interpolation conditions. It does the replacement
329     ! when certain criteria are satisfied (i.e., when ITEST = 3).
330     ! Note that SMAT = BMAT(:, 1:NPT)
331     !
332     ! See Section 8 of the NEWUOA paper.
333     
334     ! Generic modules
335     use consts_mod, only : RP, IK, ZERO, DEBUGGING, SRNLEN
336     use debug_mod, only : errstop, verisize
337     use lina_mod, only : inprod, matprod
338     
339     implicit none
340     
341     ! Inputs
342     integer(IK), intent(in) :: idz
343     real(RP), intent(in) :: fval(:)       ! FVAL(NPT)
344     real(RP), intent(in) :: ratio
345     real(RP), intent(in) :: smat(:, :)    ! SMAT(N, NPT)
346     real(RP), intent(in) :: zmat(:, :)    ! ZMAT(NPT, NPT-N-!)
347     
348     ! In-output
349     integer(IK), intent(inout) :: itest
350     real(RP), intent(inout) :: gq(:)      ! GQ(N)
351     real(RP), intent(inout) :: hq(:, :)   ! HQ(N, N)
352     real(RP), intent(inout) :: pq(:)      ! PQ(NPT)
353     ! N.B.:
354     ! GQ, HQ, and PQ should be INTENT(INOUT) instead of INTENT(OUT).
355     ! According to the Fortran 2018 standard, an INTENT(OUT) dummy
356     ! argument becomes undefined on invocation of the procedure.
357     ! Therefore, if the procedure does not define such an argument,
358     ! its value becomes undefined, which is the case for HQ and PQ
359     ! when ITEST < 3 at exit. In addition, the information in GQ is
360     ! needed for definining ITEST, so it must be INTENT(INOUT).
361     
362     ! Intermediate variables
363     integer(IK) :: n
364     integer(IK) :: npt
365     real(RP) :: fz(size(zmat, 2))
366     real(RP) :: galt(size(gq))
367     character(len = SRNLEN), parameter :: srname = 'TRYQALT'
368     
369     
370     ! Get and verify the sizes.
371     n = int(size(gq), kind(n))
372     npt = int(size(pq), kind(npt))
373     
374     if (DEBUGGING) then
375         if (n == 0 .or. npt < n + 2) then
376             call errstop(srname, 'SIZE(GQ) or SIZE(PQ) is invalid')
377         end if
378         call verisize(fval, npt)
379         call verisize(smat, n, npt)
380         call verisize(zmat, npt, int(npt - n - 1, kind(n)))
381         call verisize(hq, n, n)
382     end if
383     
384     ! In the NEWUOA paper, Powell replaces Q with Q_alt when
385     ! RATIO <= 0.01 and ||G_alt|| <= 0.1||GQ|| hold for 3 consecutive
386     ! times (eq(8.4)). But Powell's code compares ABS(RATIO) instead
387     ! of RATIO with 0.01. Here we use RATIO, which is more efficient
388     ! as observed in in Zhang Zaikun's PhD thesis (Section 3.3.2).
389     !if (abs(ratio) > 1.0e-2_RP) then
390     if (ratio > 1.0e-2_RP) then
391         itest = 0
392     else
393         galt = matprod(smat, fval)
394         if (inprod(gq, gq) < 1.0e2_RP*inprod(galt, galt)) then
395             itest = 0
396         else
397             itest = int(itest + 1, kind(itest))
398         end if
399     end if
400     
401     ! Replace Q with Q_alt when ITEST >= 3.
402     if (itest >= 3) then
403         gq = galt
404         hq = ZERO
405         fz = matprod(fval, zmat)
406         fz(1 : idz - 1) = -fz(1 : idz - 1)
407         pq = matprod(zmat, fz)
408         itest = 0
409     end if
410     
411     end subroutine tryqalt
412     
413     
414     end module update_mod
