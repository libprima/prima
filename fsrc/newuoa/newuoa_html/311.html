1     ! LINA is a module providing some basic linear algebra procedures.
2     ! To improve the performance of these procedures, especially matprod, 
3     ! one can customize their implementations, according to the resources
4     ! (hardware, e.g., cache, and libraries, e.g., BLAS) available and the
5     ! sizes of the matrices/vectors.
6     
7     ! N.B.
8     ! When implementing the code by MATLAB, Python, ..., note the following.
9     ! 1. We should following the implementation with
10     !    __USE_POWELL_ALGEBRA__ == 0, which uses matrix/vector operations
11     !    instead of loops.
12     ! 2. We should not implement the code in subroutines/functions but
13     !    write it inline, because the code is short using matrix/vector
14     !    operations, and because the overhead of subroutine/function calling
15     !    can be high in these languages.
16     
17     ! Coded by Zaikun ZHANG in July 2020.

53     subroutine r1_sym(A, alpha, x)
54     ! R1_SYM sets
55     ! A = A + ALPHA*( X*X^T ), 
56     ! where A is an NxN matrix, ALPHA is a scalar, and X is an
57     ! N-dimenional vector.
58     use consts_mod, only : RP, IK, ONE, ZERO
59     
60     #if __DEBUGGING__ == 1
61     use consts_mod, only : SRNLEN
62     use debug_mod, only : errstop, verisym
63     #endif
64     
65     implicit none
66     real(RP), intent(in) :: alpha
67     real(RP), intent(in) :: x(:)
68     real(RP), intent(inout) :: A(:, :)  ! A(SIZE(X), SIZE(X))
69     
70     integer(IK) :: n, j
71     
72     #if __DEBUGGING__ == 1
73     character(len = SRNLEN), parameter :: srname = 'R1_SYM'
74     ! Be careful with initialization!
75     ! In Fortran >=90, the initialization in the declaration implies
76     ! the "save" attribute. If the variable is not a parameter, it may
77     ! casue unwanted behavior.
78     if (size(A, 1) /= size(x) .or. size(A, 2) /= size(x)) then
79         call errstop(srname, 'SIZE(A) is invalid')
80     end if
81     #endif
82     
83     n = int(size(x), kind(n))
84     
85     #if __USE_POWELL_ALGEBRA__ == 1
86     ! Only update the LOWER TRIANGULAR part of A.
87     ! Both of the following cases are invoked in NEWUOA.
88     if (abs(alpha - ONE) > ZERO) then
89         do j = 1, n
90             A(j : n, j) = A(j : n, j) + alpha*x(j : n)*x(j)
91         end do
92     else
93         do j = 1, n
94             A(j : n, j) = A(j : n, j) + x(j : n)*x(j)
95         end do
96     end if
97     ! Set A(UPPER_TRI) = A(LOWER_TRI).
98     call symmetrize(A)
99     #else
100     ! Here we calculate the increment as ALPHA*( X*X^T ), which
101     ! is guaranteed symmetric even in floating-point arithmetic.
102     ! Both of the following cases are invoked in NEWUOA.
103     if (abs(alpha - ONE) > ZERO) then
104         do j = 1, n
105             A(:, j) = A(:, j) + alpha*( x*x(j) )
106         end do
107     else
108         do j = 1, n
109             A(:, j) = A(:, j) + x*x(j)
110         end do
111     end if
112     #endif
113     
114     #if __DEBUGGING__ == 1
115     call verisym(A, ZERO)
116     #endif
117     end subroutine r1_sym

120     subroutine r1(A, alpha, x, y)
121     ! R1 sets
122     ! A = A + ALPHA*( X*Y^T ), 
123     ! where A is an MxN matrix, ALPHA is a real scalar, X is an
124     ! M-dimenional vector, and Y is an N-dimenional vector.
125     use consts_mod, only : RP, IK, ONE, ZERO
126     
127     #if __DEBUGGING__ == 1
128     use consts_mod, only : SRNLEN
129     use debug_mod, only : errstop
130     #endif
131     
132     implicit none
133     real(RP), intent(in) :: alpha
134     real(RP), intent(in) :: x(:)
135     real(RP), intent(in) :: y(:)
136     real(RP), intent(inout) :: A(:, :)  ! A(SIZE(X), SIZE(Y))
137     
138     integer(IK) :: n, j
139     
140     #if __DEBUGGING__ == 1
141     character(len = SRNLEN), parameter :: srname = 'R1'
142     ! Be careful with initialization!
143     ! In Fortran >=90, the initialization in the declaration implies
144     ! the "save" attribute. If the variable is not a parameter, it may
145     ! casue unwanted behavior.
146     if (size(A, 1) /= size(x) .or. size(A, 2) /= size(y)) then
147         call errstop(srname, 'SIZE(A) is invalid')
148     end if
149     #endif
150     
151     n = int(size(y), kind(n))
152     
153     #if __USE_POWELL_ALGEBRA__ == 1
154     ! Both of the following cases are invoked in NEWUOA.
155     if (abs(alpha - ONE) > ZERO) then
156         do j = 1, n
157             A(:, j) = A(:, j) + alpha*x*y(j)
158         end do
159     else
160         do j = 1, n
161             A(:, j) = A(:, j) + x*y(j)
162         end do
163     end if
164     #else
165     ! Here we calculate the increment as ALPHA*( X*Y^T ) rather than
166     ! ALPHA*X*Y^T to ensure that, when X = Y, the increment is
167     ! symmetric even in floating-point arithmetic.
168     ! Both of the following cases are invoked in NEWUOA.
169     if (abs(alpha - ONE) > ZERO) then
170         do j = 1, n
171             A(:, j) = A(:, j) + alpha*( x*y(j) )
172         end do
173     else
174         do j = 1, n
175             A(:, j) = A(:, j) + x*y(j)
176         end do
177     end if
178     #endif
179     end subroutine r1

182     subroutine r2_sym(A, alpha, x, y)
183     ! R2_SYM sets
184     ! A = A + ALPHA*( X*Y^T + Y*X^T ), 
185     ! where A is an NxN matrix, X and Y are N-dimenional vectors, and
186     ! alpha is a sclalar.
187     use consts_mod, only : RP, IK, ONE, ZERO
188     
189     #if __DEBUGGING__ == 1
190     use consts_mod, only : SRNLEN
191     use debug_mod, only : errstop, verisym
192     #endif
193     
194     implicit none
195     real(RP), intent(in) :: alpha
196     real(RP), intent(in) :: x(:)
197     real(RP), intent(in) :: y(:)
198     real(RP), intent(inout) :: A(:, :)  ! A(SIZE(X), SIZE(X))
199     
200     integer(IK) :: n, j
201     
202     #if __DEBUGGING__ == 1
203     character(len = SRNLEN), parameter :: srname = 'R2_SYM'
204     if (size(x) /= size(y)) then
205         call errstop(srname, 'SIZE(X) /= SIZE(Y)')
206     end if
207     if (size(A, 1) /= size(x) .or. size(A, 2) /= size(x)) then
208         call errstop(srname, 'SIZE(A) is invalid')
209     end if
210     #endif
211     
212     n = int(size(x), kind(n))
213     
214     #if __USE_POWELL_ALGEBRA__ == 1
215     ! Only update the LOWER TRIANGULAR part of A.
216     if (abs(alpha - ONE) > ZERO) then  ! NOT invoked in NEWUOA.
217         do j = 1, n
218             A(j : n, j) = A(j : n, j) + alpha*x(j : n)*y(j) + alpha*y(j : n)*x(j)
219         end do
220     else  ! This case is invoked in NEWUOA.
221         do j = 1, n
222             A(j : n, j) = A(j : n, j) + x(j : n)*y(j) + y(j : n)*x(j)
223         end do
224     end if
225     ! Set A(UPPER_TRI) = A(LOWER_TRI).
226     call symmetrize(A)
227     #else
228     ! Here we calculate the increment as ALPHA*( X*Y^T + Y*X^T), which
229     ! is guaranteed symmetric even in floating-point arithmetic.
230     if (abs(alpha - ONE) > ZERO) then  ! NOT invoked in NEWUOA.
231         do j = 1, n
232             A(:, j) = A(:, j) + alpha*( x*y(j) + y*x(j) )
233         end do
234     else  ! This case is invoked in NEWUOA.
235         do j = 1, n
236             A(:, j) = A(:, j) + ( x*y(j) + y*x(j) )
237         end do
238     end if
239     #endif
240     
241     #if __DEBUGGING__ == 1
242     call verisym(A, ZERO)
243     #endif
244     end subroutine r2_sym

247     subroutine r2(A, alpha, x, y, beta, u, v)
248     ! R2 sets
249     ! A = A + ( ALPHA*( X*Y^T ) + BETA*( U*V^T ) ), 
250     ! where A is an MxN matrix, ALPHA and BETA are real scalars, X and
251     ! U are M-dimenional vectors, Y and V are N-dimenional vectors.
252     use consts_mod, only : RP, IK, ONE, ZERO
253     
254     #if __DEBUGGING__ == 1
255     use consts_mod, only : SRNLEN
256     use debug_mod, only : errstop
257     #endif
258     
259     implicit none
260     real(RP), intent(in) :: alpha
261     real(RP), intent(in) :: beta
262     real(RP), intent(in) :: x(:)
263     real(RP), intent(in) :: y(:)
264     real(RP), intent(in) :: u(:)          ! U(SIZE(X))
265     real(RP), intent(in) :: v(:)          ! V(SIZE(Y))
266     real(RP), intent(inout) :: A(:, :)    ! A(SIZE(X), SIZE(Y))
267     
268     integer(IK) :: n, j
269     
270     #if __DEBUGGING__ == 1
271     character(len = SRNLEN), parameter :: srname = 'R2'
272     if (size(u) /= size(x)) then
273         call errstop(srname, 'SIZE(U) /= SIZE(X)')
274     end if
275     if (size(v) /= size(y)) then
276         call errstop(srname, 'SIZE(V) /= SIZE(Y)')
277     end if
278     if (size(A, 1) /= size(x) .or. size(A, 2) /= size(y)) then
279         call errstop(srname, 'SIZE(A) is invalid')
280     end if
281     #endif
282     
283     n = int(size(y), kind(n))
284     
285     #if __USE_POWELL_ALGEBRA__ == 1
286     if (abs(alpha - ONE) > ZERO .or. abs(beta - ONE) > ZERO) then
287         ! This case is NOT invoked in NEWUOA.
288         do j = 1, n
289             A(:, j) = A(:, j) + alpha*x*y(j) + beta*u*v(j)
290         end do
291     else  ! This case is invoked in NEWUOA.
292         do j = 1, n
293             A(:, j) = A(:, j) + x*y(j) + u*v(j)
294         end do
295     end if
296     #else
297     if (abs(alpha - beta) > ZERO) then  ! NOT invoked in NEWUOA.
298         do j = 1, n
299             A(:, j) = A(:, j) + (alpha*(x*y(j)) + beta*(u*v(j)))
300         end do
301     else if (abs(alpha - ONE) > ZERO) then  ! NOT invoked in NEWUOA.
302     ! Here update A to A + ALPHA*( X*Y^T + U*V^T ), a semmetric update
303     ! when X = V and Y = U, even in floating-point arithmetic.
304         do j = 1, n
305             A(:, j) = A(:, j) + alpha*( x*y(j)  + u*v(j) )
306         end do
307     else  ! This case is invoked in NEWUOA.
308     ! Here update A to A + ( X*Y^T + U*V^T ), a semmetric update
309     ! when X = V and Y = U, even in floating-point arithmetic.
310         do j = 1, n
311             A(:, j) = A(:, j) + ( x*y(j) + u*v(j) )
312         end do
313     end if
314     #endif
315     end subroutine r2

488     subroutine grota(A, i, j, k)
489     ! GROTA sets A = A*G, 
490     ! where G is the Givens rotation that
491     !    [A(K, 1), ..., A(K, I), ..., A(K, J), ..., A(K, N)] * G
492     !  = [A(K, 1), ..., R   , ..., 0   , ..., A(K, N)]
493     ! with R = SQRT(A(K, I)^2 + A(K, J)^2).
494     ! Indeed, the [(I, I), (I, J); (J, I), (J, J)] block of G is
495     ! [C, -S; S, C], where C = A(K, I)/R, S = A(K, J)/R, and the other
496     ! entries are 1 (diagonal) or 0. Therefore, A*G is identical to A
497     ! except for columns I and J:
498     ! new column I =  C*A(:, I) + S*A(:, J)
499     ! new column J = -S*A(:, I) + C*A(:, J)
500     
501     use consts_mod, only : IK, RP, ZERO
502     
503     #if __DEBUGGING__ == 1
504     use consts_mod, only : SRNLEN
505     use debug_mod, only : errstop
506     #endif
507     
508     implicit none
509     integer(IK), intent(in) :: i
510     integer(IK), intent(in) :: j
511     integer(IK), intent(in) :: k
512     real(RP), intent(inout) :: A(:, :)
513     
514     real(RP) :: c, s, r, Ai(size(A, 1))
515     
516     #if __DEBUGGING__ == 1
517     character(len = SRNLEN), parameter :: srname = 'GROTA'
518     if (i == j .or. min(i, j) < 1 .or. max(i, j) > size(A, 2)) then
519         call errstop(srname, 'I or J is invalid')
520     end if
521     if (k < 1 .or. k > size(A, 1)) then
522         call errstop(srname, 'K is invalid')
523     end if
524     #endif
525     
526     if (abs(A(k, j)) >  ZERO) then
527         c = A(k, i)
528         s = A(k, j)
529         ! One may use R = HYPOT(C, s), but it did not work better
530         ! in our experiments on CUTEst problems.
531         r = sqrt(c**2 + s**2)
532         c = c/r
533         s = s/r
534         Ai = A(:, i)
535         A(:, i) =  c*Ai + s*A(:, j)
536         A(:, j) = -s*Ai + c*A(:, j)
537         A(k, j) = ZERO
538     
539     #if __USE_POWELL_ALGEBRA__ == 0
540         A(k, i) = r  ! A(K, I) = R in precise arithmetic
541     else if (A(k, i) < ZERO) then
542         ! In this case, C = -1, S = 0. Thus it is reasonbale to set
543         ! the following. Doing this will ensure the continuity of this
544         ! subroutine as an operator.
545             A(:, i) = -A(:, i)
546             A(:, j) = -A(:, j)
547     #endif
548     end if
549     end subroutine grota

552     subroutine symmetrize(A)
553     ! SYMMETRIZE(A) symmetrizes A.
554     ! Here, we assume that A is a matrix that is symmetric in precise
555     ! arithmetic, and its asymetry comes only from rounding errors.
556     
557     #if __USE_POWELL_ALGEBRA__ == 1
558     use consts_mod, only : RP, IK
559     #else
560     use consts_mod, only : RP, HALF
561     #endif
562     
563     #if __DEBUGGING__ == 1
564     use consts_mod, only : ZERO, SRNLEN
565     use debug_mod, only : errstop, verisym
566     #endif
567     
568     implicit none
569     real(RP), intent(inout) :: A(:, :)
570     
571     #if __USE_POWELL_ALGEBRA__ == 1
572     integer(IK) :: j
573     #endif
574     
575     #if __DEBUGGING__ == 1
576     character(len = SRNLEN), parameter :: srname = 'SYMMETRIZE'
577     if (size(A, 1) /= size(A, 2)) then
578         call errstop(srname, 'A is not square')
579     end if
580     #endif
581     
582     #if __USE_POWELL_ALGEBRA__ == 1
583     ! A is symmetrized by setting A(UPPER_TRI) = A(LOWER_TRI).
584     do j = 1, int(size(A, 1), kind(j))
585         A(1 : j - 1, j) = A(j, 1 : j - 1)
586     end do
587     #else
588     A = A + transpose(A)
589     A = A*HALF
590     #endif
591     
592     #if __DEBUGGING__ == 1
593     call verisym(A, ZERO)
594     #endif
595     end subroutine symmetrize

598     function xpy_dot_z (x, y, z) result(t)
599     ! t = (x + y)'*z
600     
601     use consts_mod, only : RP
602     
603     #if __USE_POWELL_ALGEBRA__ == 1
604     use consts_mod, only : IK, ZERO
605     #endif
606     
607     #if __DEBUGGING__ == 1
608     use consts_mod, only : SRNLEN
609     use debug_mod, only : errstop
610     #endif
611     implicit none
612     
613     real(RP), intent(in) :: x(:)
614     real(RP), intent(in) :: y(:)
615     real(RP), intent(in) :: z(:)
616     real(RP) :: t
617     
618     #if __USE_POWELL_ALGEBRA__ == 1

773     subroutine calquad(d, gq, hq, pq, x, xpt, vquad)
774     ! CALQUAD calculates
775     ! VQUAD = Q(X + D) - Q(X)
776     ! with Q being the quadratic function defined via (GQ, HQ, PQ) by
777     ! Q(Y) = <Y, GQ> + 0.5*<Y, HESSIAN*Y>, 
778     ! where HESSIAN consists of an explicit part HQ and an implicit
779     ! part PQ in the following way:
780     ! HESSIAN = HQ + sum_K=1^NPT PQ(K)*(XPT(:, K)*XPT(:, K)^T) .
781     
782     use consts_mod, only : RP, HALF
783     
784     #if __USE_POWELL_ALGEBRA__ == 1 || __DEBUGGING__ == 1
785     use consts_mod, only : IK
786     #endif
787     
788     #if __USE_POWELL_ALGEBRA__ == 1
789     use consts_mod, only : ZERO
790     #endif
791     
792     #if __DEBUGGING__ == 1
793     use consts_mod, only : SRNLEN
794     use debug_mod, only : errstop, verisize
795     #endif
796     
797     implicit none
798     
799     ! Inputs
800     real(RP), intent(in) :: d(:)      ! D(N)
801     real(RP), intent(in) :: gq(:)     ! GQ(N)
802     real(RP), intent(in) :: hq(:, :)  ! HQ(N, N)
803     real(RP), intent(in) :: pq(:)     ! PQ(NPT)
804     real(RP), intent(in) :: x(:)      ! X(N)
805     real(RP), intent(in) :: xpt(:, :) ! XPT(N, NPT)
806     
807     ! Output
808     real(RP), intent(out) :: vquad
809     
810     ! Intermediate variable
811     real(RP) :: s(size(x))
812     
813     #if __USE_POWELL_ALGEBRA__ == 1
814     real(RP) :: w(size(pq)), t
815     integer(IK) :: i, ih, j
816     #endif
817     
818     #if __DEBUGGING__ == 1
819     integer(IK) :: n, npt
820     character(len = SRNLEN), parameter :: srname = 'CALQUAD'
821     n = int(size(xpt, 1), kind(n))
822     npt = int(size(xpt, 2), kind(npt))
823     if (n == 0 .or. npt < n + 2) then
824         call errstop(srname, 'SIZE(XPT) is invalid')
825     end if
826     call verisize(d, n)
827     call verisize(x, n)
828     call verisize(gq, n)
829     call verisize(hq, n, n)
830     call verisize(pq, npt)
831     #endif
832     
833     #if __USE_POWELL_ALGEBRA__ == 1
834     s = x + d
835     ! First order term and explicit second order term
836     vquad = ZERO
837     ih = 0
838     do j = 1, int(size(d), kind(j))
839         vquad = vquad + d(j)*gq(j)
840         do i = 1, j
841             ih = int(ih + 1, kind(ih))
842             t = d(i)*s(j) + d(j)*x(i)
843             if (i == j) then
844                 t = HALF*t
845             end if
846             vquad = vquad + t*hq(i, j)
847         end do
848     end do
849     !----------------------------------------------------------------------!
850     ! Powell's original code calculates W externally and pass it as an
851     ! input to CALQUAD. Here W is calculated internally.
852     w = matprod(d, xpt) !--------------------------------------------!
853     w = w*(HALF*w + matprod(x, xpt)) !-------------------------------!
854     !----------------------------------------------------------------------!
855     ! Implicit second order term
856     do i = 1, int(size(pq), kind(i))
857         vquad = vquad + pq(i)*w(i)
858     end do
859     #else
860     ! The order of calculation seems quite important. The following
861     ! order seems to work well.
862     ! 1st order term
863     vquad = inprod(d, gq)
864     s = HALF*d + x  ! Different from the above version.
865     ! implicit 2nd-order term
866     vquad = vquad + sum(pq * (matprod(s, xpt) * matprod(d, xpt)))
867     ! explicit 2nd-order term
868     vquad = vquad + inprod(s, matprod(hq, d))
869     ! The following implementations do not work as well as the above one.
870     !vquad = vquad + inprod(d, matprod(hq, s))
871     !vquad = vquad + sum(hq * outprod(s, d))
872     !vquad = vquad + HALF*(inprod(d, matprod(hq, s)) + inprod(s, matprod(hq, d)))
873     #endif
874     end subroutine calquad
